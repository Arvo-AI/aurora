export interface LokiStatus {
  connected: boolean;
  baseUrl?: string;
  authType?: string;
  tenantId?: string;
  error?: string;
}

export interface LokiConnectPayload {
  baseUrl: string;
  authType: "bearer" | "basic" | "none";
  token?: string;
  username?: string;
  password?: string;
  tenantId?: string;
}

export interface WebhookUrlResponse {
  webhookUrl: string;
  instructions: string[];
}

export const lokiService = {
  getStatus(): Promise<LokiStatus | null>;
  connect(payload: LokiConnectPayload): Promise<LokiStatus>;
  getWebhookUrl(): Promise<WebhookUrlResponse>;
}

<!-- Backend /loki/connect request shape (from loki_routes.py) -->
POST /loki/connect expects:
  { baseUrl: string, authType: "bearer"|"basic"|"none", token?: string, username?: string, password?: string, tenantId?: string }
Returns on success:
  { success: true, baseUrl: string, authType: string, tenantId: string|null, labelsCount: number }

GET /loki/alerts/webhook-url returns:
  { webhookUrl: string, instructions: string[] }
  Instructions contain both "Option A: Configure Loki Ruler with Alertmanager" and "Option B: Configure Grafana Alerting contact point" sections.

**client/src/app/loki/auth/page.tsx:**

`"use client"` directive. Import:

- `copyToClipboard` from `@/lib/utils`

Export default function `LokiAuthPage()`. State variables:
- `baseUrl: string` (default "")
- `authType: "bearer" | "basic" | "none"` (default "bearer")
- `token: string` (default "")
- `username: string` (default "")
- `password: string` (default "")
- `tenantId: string` (default "")
- `status: LokiStatus | null` (default null)
- `loading: boolean` (default false)
- `isInitialLoading: boolean` (default true)
- `webhookUrl: string | null` (default null)
- `instructions: string[]` (default [])
- `webhookLoading: boolean` (default false)
- `webhookError: string | null` (default null)
- `copiedField: "url" | null` (default null)

Functions (follow Netdata pattern):

`loadStatus` async: calls `lokiService.getStatus()`, sets status, if connected calls `loadWebhookUrl()`, catches errors, finally sets isInitialLoading false.

`loadWebhookUrl` async: sets webhookLoading true, webhookError null, calls `lokiService.getWebhookUrl()`, sets webhookUrl and instructions from response. On error: sets webhookError, shows toast. Finally sets webhookLoading false.

`useEffect(() => { loadStatus(); }, [])` on mount.

`handleConnect` async (event: React.FormEvent<HTMLFormElement>): preventDefault, setLoading true. Build payload: `{ baseUrl, authType, ...(authType === "bearer" ? { token } : {}), ...(authType === "basic" ? { username, password } : {}), ...(tenantId ? { tenantId } : {}) }`. Call `lokiService.connect(payload)`, set status, show success toast "Loki connected! Now configure the webhook below.", call loadWebhookUrl. Set localStorage "isLokiConnected" = "true", dispatch "providerStateChanged" event. On error: show destructive toast. Finally: setLoading false, clear sensitive fields (setToken(""), setPassword("")).

`handleDisconnect` async: setLoading true. Fetch DELETE `/api/connected-accounts/loki` with credentials include. On success (ok or 204): reset all state (status to {connected: false}, webhookUrl to null, instructions to [], clear form fields), show success toast, remove localStorage "isLokiConnected", dispatch "providerStateChanged". On error: show destructive toast. Finally setLoading false.

`handleCopy` function (text: string): calls copyToClipboard, sets copiedField to "url", shows "Copied" toast, setTimeout to clear copiedField after 2000ms.

Render structure (wrapped in ConnectorAuthGuard with connectorName="Loki"):
- Container: `div className="container mx-auto py-8 px-4 max-w-3xl"`
- Header: h1 "Loki Integration", p "Connect Grafana Loki for log aggregation and alert webhooks"
- If isInitialLoading: show "Loading..." centered text
- If !status?.connected: render `<LokiConnectionStep>` with all form state props and handlers
- If status?.connected:
  - If webhookLoading: show "Loading webhook configuration..." centered text
  - If webhookError: show error text + Retry button (onClick loadWebhookUrl)
  - If webhookUrl and status: render `<LokiWebhookStep>` with status, webhookUrl, instructions, copiedField, onCopyUrl, onDisconnect, loading props
